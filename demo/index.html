Integration

1. Install

npm install @duyet/fonts

2. Import

import '@duyet/fonts/fonts/duyet-serif/index.css';

Tailwind CSS

// tailwind.config.js
module.exports = {
  theme: {
    extend: {
      fontFamily: {
        serif: ['"Duyet Serif"', 'serif'],
      },
    },
  },
}
